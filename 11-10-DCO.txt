COPY
(
select
    product_kode,
    tanggal_order,
    state as "Status Pengiriman",
    reff as "Refference",
    card_name,
    city,
    zipcode,
    rpx_delivery_date as "Delivery Date", 
    rpx_retur_status as "Retur Status",
    rpx_received_by as "Received By", 
    jenis_product,
    rpx_rcv_relation as "Hubungan Penerima", 
    kode_kurir
from bmri_surat
where product_kode='DCO' 
    and DATE_PART('YEAR',tanggal_order)='2024'
    and DATE_PART('MONTH',tanggal_order) in ('09')
    and state='delivered'
)
TO '/home/trackinghubadmin/CEKDATA/20241011_DCO_DELIVERED.csv' With CSV DELIMITER ',' HEADER;
