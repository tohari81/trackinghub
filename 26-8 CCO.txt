COPY
(
select
    reff,
    card_name,
    tanggal_order,
    rpx_delivery_date as "Delivery Date", 
    state as "Status Pengiriman",
    rpx_received_by as "Received By", 
    rpx_rcv_relation as "Relation", 
    kode_kurir
from bmri_surat
where product_kode='CCO' 
    and DATE_PART('YEAR',tanggal_order)='2024'
    and DATE_PART('MONTH',tanggal_order) in ('05','06','07')
    and state='delivered'
)
TO '/home/trackinghubadmin/CEKDATA/20240826_CCO_DELIVERED.csv' With CSV DELIMITER ',' HEADER;
